project(main_bom)

set(CMAKE_CXX_STANDARD 14)

# Define the cross compiler locations
SET(CMAKE_C_COMPILER   /media/david/DATA_2/school/3.3/Project/codes/tools/arm-bcm2708/arm-rpi-4.9.3-linux-gnueabihf/bin/arm-linux-gnueabihf-g++)
